Server-side changes only. Extension-side changes in pi-subagents-http spec.

## Prerequisite: Submodule migration


## Decisions Log

| Question | Decision | Rationale |
|----------|----------|-----------|
| SSE server library | `better-sse` | 825 stars, zero deps, built-in channels, auto-cleanup on disconnect. Framework-agnostic via request.raw/reply.raw. |
| OpenAI model ID | `pi/{agentName}` | Namespaced to avoid collision in Open WebUI dropdown. Matches OpenClaw pattern. |
| OpenAI multi-turn | Stateless — last user message = task | Each request is independent task. System messages become context. Prior messages ignored. |
| OpenAI streaming | Status messages during execution, final output at end | Stream "[Working... turn 3]" as visible content. No raw LLM noise. SSE comment heartbeats every 15s. |
| OpenAI architecture | Routes in existing server.ts | Reuses processInvocation(), run tracking, session management. No separate gateway. |
| OpenAI timeout | `BRIDGE_TIMEOUT_MS` | Already configured per agent (1800s planner, 600s workers). SSE heartbeats prevent proxy timeouts. |

## Enhancement 1: SSE Event Bus + `/events` Endpoint

### New dependency

```json
{ "dependencies": { "better-sse": "^0.16.0" } }
```

### Changes to `server.ts`

**Event bus** — thin layer on existing `trackRun()`:

```typescript
import { createSession, createChannel } from "better-sse";

const sseChannel = createChannel();

function emitRunEvent(type: string, runId: string, data: Record<string, any>) {
  sseChannel.broadcast({ runId, ...data }, `run:${type}`);
}
```

Wire into existing code points:

```typescript
// In trackRun() — after updating runs Map:
function trackRun(runId: string, data: Record<string, any>) {
  const prev = runs.get(runId);
  runs.set(runId, { ...prev, ...data });
  // ... existing cleanup ...

  // Emit SSE events on state transitions
  if (data.status === "queued") emitRunEvent("accepted", runId, { state: "queued" });
  if (data.status === "running" && prev?.status !== "running") emitRunEvent("progress", runId, { state: "running" });
  if (data.status === "completed") emitRunEvent("completed", runId, { state: "completed", output: data.output, usage: data.usage, durationMs: /* ... */ });
  if (data.status === "failed") emitRunEvent("failed", runId, { state: "failed", error: data.error });
  if (data.status === "cancelled") emitRunEvent("cancelled", runId, { state: "cancelled" });
}

// In handleSessionEvent() on turn_end — emit progress:
if (event.type === "turn_end") {
  // ... existing usage tracking ...
  emitRunEvent("progress", requestId, { state: "running", turnCount: usageByTurn.length });
}
```

**SSE route:**

```typescript
app.get("/events", async (req, reply) => {
  const session = await createSession(req.raw, reply.raw);
  sseChannel.register(session);
  // better-sse auto-deregisters on disconnect

  // Heartbeat every 15s keeps connection alive through reverse proxies
  const heartbeat = setInterval(() => {
    try { session.push({ ts: Date.now(), activeRuns: activeCount }, "heartbeat"); }
    catch { clearInterval(heartbeat); }
  }, 15_000);

  req.raw.on("close", () => clearInterval(heartbeat));
});
```

**Response headers** set by better-sse automatically:
- `Content-Type: text/event-stream`
- `Cache-Control: no-cache`
- `Connection: keep-alive`

Add `X-Accel-Buffering: no` for nginx compatibility:
```typescript
reply.raw.setHeader("X-Accel-Buffering", "no");
```

### Config

Environment variables:
```
OPENAI_COMPAT_ENABLED=true        # mount /v1 routes
OPENAI_COMPAT_API_KEY=            # empty = no auth (local Docker)
```

Read in server.ts:
```typescript
const openaiEnabled = process.env.OPENAI_COMPAT_ENABLED === "true";
const openaiApiKey = process.env.OPENAI_COMPAT_API_KEY || "";
```

```typescript
app.get("/v1/models", async (req, reply) => {
```

### `POST /v1/chat/completions`

Always responds with SSE streaming to prevent proxy/client timeouts on long-running agents.

```typescript
app.post("/v1/chat/completions", async (req, reply) => {
  if (!openaiEnabled) return reply.code(404).send({ error: "not_found" });

  // Optional bearer auth
  if (openaiApiKey) {
    const auth = req.headers.authorization;
    if (!auth || auth !== `Bearer ${openaiApiKey}`) {
      return reply.code(401).send({ error: { message: "unauthorized", type: "auth_error" } });
    }
  }

  const body = req.body as any;
  const messages = body.messages || [];

  // Extract last user message as task
  const lastUserMsg = [...messages].reverse().find((m: any) => m.role === "user");
  if (!lastUserMsg) {
    return reply.code(400).send({ error: { message: "no user message in messages array", type: "invalid_request_error" } });
  }

  // System messages become context
  const systemMsgs = messages.filter((m: any) => m.role === "system").map((m: any) => m.content);
  const context = systemMsgs.length > 0 ? systemMsgs.join("\n\n") : undefined;
  const task = typeof lastUserMsg.content === "string"
    ? lastUserMsg.content
    : JSON.stringify(lastUserMsg.content);

  const requestId = randomUUID().replace(/-/g, "");
  const completionId = `chatcmpl-${requestId}`;
  const modelId = `pi/${AGENT_NAME}`;
  const created = Math.floor(Date.now() / 1000);

  // Queue check
  if (activeCount >= MAX_CONCURRENT && waitQueue.length >= QUEUE_MAX_DEPTH) {
    return reply.code(429).send({ error: { message: "agent queue full", type: "rate_limit_error" } });
  }

  // Start SSE stream immediately
  reply.raw.writeHead(200, {
    "Content-Type": "text/event-stream",
    "Cache-Control": "no-cache, no-transform",
    "Connection": "keep-alive",
    "X-Accel-Buffering": "no",
  });

  // Helper to write OpenAI-format SSE chunks
  const sendChunk = (content: string | null, finishReason: string | null = null) => {
    const chunk = {
      id: completionId, object: "chat.completion.chunk", created, model: modelId,
      choices: [{
        index: 0,
        delta: content !== null ? { content } : {},
        finish_reason: finishReason,
      }],
    };
    reply.raw.write(`data: ${JSON.stringify(chunk)}\n\n`);
  };

  // Initial role chunk
  reply.raw.write(`data: ${JSON.stringify({
    id: completionId, object: "chat.completion.chunk", created, model: modelId,
    choices: [{ index: 0, delta: { role: "assistant", content: "" }, finish_reason: null }],
  })}\n\n`);

  // Heartbeat every 15s (SSE comment — invisible to clients)
  const heartbeat = setInterval(() => {
    reply.raw.write(`: heartbeat\n\n`);
  }, 15_000);

  // Status updates every 30s (visible content)
  let statusCount = 0;
  const requestStart = Date.now();
  const statusTimer = setInterval(() => {
    statusCount++;
    const elapsed = Math.round((Date.now() - requestStart) / 1000);
    sendChunk(`[Working... ${elapsed}s elapsed]\n`);
  }, 30_000);

  // Track and execute
  metrics.requests_total++;
  metrics.requests_active++;
  metrics.last_request_at = new Date().toISOString();

  const abortController = new AbortController();
  activeAborts.set(requestId, abortController);
  trackRun(requestId, {
    status: "queued", startedAt: new Date().toISOString(), startedAtMs: Date.now(),
    correlationId: requestId, output: null, error: null, usage: null, turnCount: 0,
  });

  try {
    await acquireSlot();
    await processInvocation(
      { task, context, correlationId: requestId },
      requestId, requestStart, abortController,
    );
  } catch (err: any) {
    clearInterval(heartbeat);
    clearInterval(statusTimer);
    sendChunk(`\n\n[Error: ${err.message}]`);
    sendChunk(null, "stop");
    reply.raw.write(`data: [DONE]\n\n`);
    reply.raw.end();
    return;
  }

  clearInterval(heartbeat);
  clearInterval(statusTimer);

  // Stream final output
  const run = runs.get(requestId);
  if (run?.status === "completed" && run.output) {
    sendChunk(run.output);
  } else {
    sendChunk(`\n\n[Agent ${run?.status || "unknown"}: ${run?.error || "no output"}]`);
  }

  sendChunk(null, "stop");

  // Usage chunk (if requested by client)
  if (body.stream_options?.include_usage) {
    const u = run?.usage || {};
    reply.raw.write(`data: ${JSON.stringify({
      id: completionId, object: "chat.completion.chunk", created, model: modelId,
      choices: [],
      usage: {
        prompt_tokens: u.inputTokens || 0,
        completion_tokens: u.outputTokens || 0,
        total_tokens: (u.inputTokens || 0) + (u.outputTokens || 0),
      },
    })}\n\n`);
  }

  reply.raw.write(`data: [DONE]\n\n`);
  reply.raw.end();
});
```
